	can_select = {
		any_owned_province = {
			OR = {
				has_terrain = mountain

				has_terrain = glacier
			}
			OR = {
				culture = tibetan
				culture = nepali

			OR = {
				region = tibet_region

				area = baisi_rajya_area
			}
		}
	}
	on_granted = {
		custom_tooltip = estate_hill_clans_enlist_the_sherpas_tt
	}
	on_granted_province = {
		if = {
			limit = {
				has_terrain = mountain
				OR = {
					culture = tibetan
					culture = nepali
				}
				OR = {
					region = tibet_region
					area = nepal_area
					area = baisi_rajya_area
				}
			}
			add_province_modifier = {
				name = enlist_the_sherpas_mountains
				duration = -1
			}
		}
		if = {
			limit = {
				has_terrain = highlands
				OR = {
					culture = tibetan
					culture = nepali
				}
				OR = {
					region = tibet_region
					area = nepal_area
					area = baisi_rajya_area
				}
			}
			add_province_modifier = {
				name = enlist_the_sherpas_highlands
				duration = -1
			}
		}
		if = {
			limit = {
				has_terrain = glacier
				OR = {
					culture = tibetan
					culture = nepali
				}
				OR = {
					region = tibet_region
					area = nepal_area
					area = baisi_rajya_area
				}
			}
			add_province_modifier = {
				name = enlist_the_sherpas_glacial
				duration = -1
			}
		}
		if = {
			limit = {
				has_terrain = hills

					culture = tibetan
					culture = nepali
				}
				OR = {
					region = tibet_region
					area = nepal_area
					area = baisi_rajya_area
				}
			}
			add_province_modifier = {
				name = enlist_the_sherpas_hills
				duration = -1
			}
		}
	}
	on_revoked = {
		custom_tooltip = estate_hill_clans_enlist_the_sherpas_revoke_tt
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = enlist_the_sherpas_glacial
			}
			remove_province_modifier = enlist_the_sherpas_glacial
		}
		if = {
			limit = {
				has_province_modifier = enlist_the_sherpas_highlands
			}
			remove_province_modifier = enlist_the_sherpas_highlands
		}

			}
			remove_province_modifier = enlist_the_sherpas_mountains
		}
	}
