upgrade_center_of_trade = {
	if = {
		limit = {
			NOT = {
				$province$ = {
					province_has_center_of_trade_of_level = 3
				}
			}
		}
		$province$ = {
			add_center_of_trade_level = 1
		}
	}
	else = {
		custom_tooltip = ME_add_1_years_of_trade_income
		hidden_effect = {
			add_years_of_trade_income = {
				years = 1
			}
		}
	}
}
