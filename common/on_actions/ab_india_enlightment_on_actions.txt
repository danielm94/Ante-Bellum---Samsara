on_culture_promoted = {
	if = {
		limit = {
			has_government_mechanic = state_enlightenment_mechanic
		}
		add_government_power = {
			mechanic_type = state_enlightenment_mechanic
			power_type = enlightenment_power
			value = 1
		}
	}
}

on_culture_demoted = {
	if = {
		limit = {
			has_government_mechanic = state_enlightenment_mechanic
		}
		add_government_power = {
			mechanic_type = state_enlightenment_mechanic
			power_type = enlightenment_power
			value = -1
		}
	}
}
