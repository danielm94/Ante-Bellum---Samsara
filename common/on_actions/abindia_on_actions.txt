		}
	}

			is_part_of_hre = yes

			}
		}

			id = hre_elector_inherit.001
		}
	}

	}

		limit = {
			FROM = {
				has_reform = irish_highking_opp
			}
		}
		FROM = { save_event_target_as = hk_ire }
		# howard: it is possible to annex a country via a subject
		# so we need to redirect the event to the overlord.
		if = {
			limit = {
				is_subject = yes
